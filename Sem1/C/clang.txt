int fact(int num1)
{
    
    if(num1<1)
       return 1;
	
	
	return num1*fact(num1-1);
    
}
int main()
{
    int num1;
    printf("Enter your digit:");
    scanf("%d",&num1);
    printf("factorial is: %d",fact(num1));
}
